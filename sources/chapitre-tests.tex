\begin{frame}{Présentation}
	Tous les programmes réels contiennent des \emph{bugs} \\(erreurs, défauts)
	\begin{itemize}
		\item Parfois même \emph{beaucoup}
		\item \emph{Inacceptable}
		\begin{itemize}
			\item \emph{Inconfort} pour l'utilisateur
  			\item \emph{Perte} de temps, d'argent, de données, de matériel
  			\item Voire \emph{danger} pour la vie humaine
  		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Présentation}
	Rappel des types d'erreurs
	\begin{itemize}
		\item À la \emph{compilation}
		\item À l'\emph{exécution}, le programme \emph{s'arrête}
		\item À l'\emph{exécution}, le programme fournit une \emph{mauvaise réponse}
	\end{itemize}
	\medskip
	On pourrait aussi parler d'autres défauts
	\begin{itemize}
  		\item Trop \emph{lent}
  		\item Trop gourmand en \emph{mémoire}
  	\end{itemize}
\end{frame}

\full[bluepigment]{
	\large\bf\color{white}
	\begin{center}
		\og\,J'ai fait tourner le programme;
		\\il fonctionne très bien.\,\fg
	\end{center}
}

\note{
	\emph{Insuffisant !} Il compile et tourne dans les cas les plus courants. Mais :
  	\begin{itemize}
  		\item Cas particuliers
  		\item Comportement face à une défaillance de l'environnement
  		\item Comportement face à une utilisation non conforme
  	\end{itemize}
}

\begin{frame}{Présentation}
	Pour produire un logiciel \emph{sans bug} il faut
  	\begin{itemize}
  		\item Suivre une \emph{méthodologie} éprouvée 
    	\begin{itemize}
    		\item Pour produire une première version avec peu de bugs 
    			\\(cf. \textit{Analyse})
    	\end{itemize}
  		\item \emph{Tester}, \emph{tester} et \dots \emph{ tester} encore !
    	\begin{itemize}
    		\item Pour détecter ceux qui restent
    		\item Besoin d'outils pour nous aider
      		\begin{itemize}
      			\item Les plus faciles / rapides possible
      		\end{itemize}
    	\end{itemize}
  	\end{itemize}
\end{frame}

\subsection{Les tests}

\begin{frame}[fragile]{Les tests}
	\textit{Il existe plusieurs sortes de tests\,: 
	unitaires, d'intégration, fonctionnels, non-régression, \dots}
	\vspace{.4cm}
	\par Nous nous intéressons aux tests \emph{unitaires} 
	\vspace{1.4cm}
	\par\textit{Pour en savoir plus} : \\
	{\small \url{http://fr.wikipedia.org/wiki/Test_(informatique)}}
\end{frame}

\full[bluepigment]{
	\color{white}
	\textbf{Test unitaire:}
	\par Procédure permettant de tester le bon fonctionnement d'un module 
	(une méthode)
}

\note{
	Tests \emph{unitaires} : test de \emph{chaque méthode} 
  	\begin{itemize}
  		\item Fait-elle ce qu'elle est censée faire ?
  		\item C'est défini par la \textit{spécification} (documentation)
  		\item Idée : Si chaque méthode est correcte \\$\longrightarrow$ 
  			le tout est correct
  		\item Pas forcément suffisant 
		\\(ex: ne teste pas la performance)
		\\$\longrightarrow$ d'autres types de tests existent
	\end{itemize}
}

\begin{frame}{Plan de tests}
	\begin{itemize}
		\item Planifier les tests
		\item Choisir les cas intéressants / judicieux
		\item Se baser sur les erreurs fréquentes
	\end{itemize}
	\bigskip
	\begin{flushright}
		\emph{Expérience}
	\end{flushright}
\end{frame}

\note{
	\begin{itemize}
		\item Besoin d'un \emph{plan} reprenant les tests à effectuer
	  	\begin{itemize}
  			\item Quelles \emph{valeurs de paramètres} ?
  			\item Quel est le \emph{résultat attendu} ?
  		\end{itemize}
		\item Préparé pendant que l'on code (ou même avant et éventuellement 
			par une autre personne)
		\item Permet de s'assurer que l'on teste \emph{tous les cas}
	\end{itemize}
}

\note{
	On ne peut pas tester toutes les valeurs possibles
	\begin{itemize}
		\item Choisir des valeurs \emph{représentatives}
	 	\begin{itemize}
  			\item Cas général / particuliers
  			\item Valeurs limites
  		\end{itemize}
		\item Il faut imaginer les cas qui pourraient mettre en évidence un 
			défaut de la méthode
	\end{itemize}
}

\note{
	On s'inspire des erreurs les plus fréquentes en programmation
  	\begin{itemize}
  		\item On commence/arrête trop tôt/tard une boucle
  		\item On initialise mal une variable
  		\item Dans un test, on se trompe entre $<$ et $\leq$
  		\item Dans un test, on se trompe entre $ET$ et $OU$
  		\item \dots
  	\end{itemize}
	C'est un savoir-faire $\Longrightarrow$ Importance de l'expérience
}

\begin{frame}{Plan de tests - Exemple}
	\emph{Exemple}
	\medskip
	\\\java{public static int max(int[] nbs) {...}}
	\\qui calcule la valeur maximale trouvée dans le tableau
	\begin{itemize}
		\item Que tester en plus du cas général ?
  			\begin{itemize}
  			\item Le maximum est la première/dernière valeur
  			\item Présence de nombres négatifs
  			\item Cas d'un tableau de taille 0
  			\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Plan de tests - Exemple}
	\small
	\begin{center}
		\textbf{Plan de tests de la méthode \textit{max}}
	\\\bigskip
	\begin{tabular}{|c|l|c|p{5cm}|}

\hline
1 & $1,3,0$ & $3$ & cas général\\

\begin{frame}{Plan de tests}
	\emph{Quand} tester ? Le plus \emph{souvent} possible 
	\begin{itemize}
		\item Erreur plus facile à identifier/corriger
		\item Idéalement après chaque méthode écrite
	\end{itemize}
	\medskip
	\emph{Que} tester ? \emph{Tout}
	\begin{itemize}
	\item Le nouveau code peut mettre en évidence un problème dans le 
		code ancien (\emph{régression})
	\end{itemize}
\end{frame}

\subsection{JUnit}

\begin{frame}{JUnit}
	\emph{Comment} tester ?
	\begin{itemize}
		\item Pas à la main : intenable
		\item Besoin d'un outil \emph{automatisé}
	\end{itemize}
\end{frame}

\note{
	{\small \sigle{JUnit}} : outil pour automatiser les tests unitaires
	\begin{itemize}
		\item Le programmeur fournit les tests,
		\item \sigle{JUnit} \emph{exécute} tous les tests et
		\item établit un \emph{rapport} détaillant les problèmes
	\end{itemize}
}

\begin{frame}{JUnit - En pratique}
La classe de test contient \emph{une méthode} de test \emph{par cas}
  \begin{itemize}
  \item Autonome (ne reçoit rien ne retourne rien)
  \item Contient des \emph{affirmations}
    \begin{itemize}
    \item Appel de la méthode à tester
    \item \emph{Comparaison} entre le résultat \emph{attendu} et le résultat \emph{obtenu}
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{JUnit - En pratique}
	\emph{Exemple}
	\begin{Java}
@Test
public void max_cas1() {
  int[] nbs = [1,3,0];
  assertEquals( 3, MaClasse.max(nbs) );
}
	\end{Java}
	{\normalsize
	\begin{itemize}
		\item Reconnu comme un test unitaire grâce à l'\emph{annotation} 
			\java{@Test}
		\item Pas de \java{static}
		\item \java{assertEquals} vérifie que les 2 valeurs sont identiques
		\item \java{assertTrue(val)}, \java{assertFalse(val)}, \dots
	\end{itemize}
}
\end{frame}

\begin{frame}[fragile]{JUnit - En pratique}
	Comment \emph{lancer} les tests ?
	\begin{Java}
java org.junit.runner.JUnitCore mon.paquet.MaClasseTest
	\end{Java} %$
	\medskip
	Résultats\,:
	\begin{itemize}
		\item Nombre de tests effectués / réussis
		\item Détail sur les tests ratés
	  	\begin{itemize}
  			\item Nom du test
  			\item Résultat obtenu comparé au résultat attendu
  		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}[fragile]{JUnit - Exemple}
	\emph{Exemple}
\begin{Java}
package be.he2b.esi.java1;
public class Outil {
  public static int max(int[] nbs) {
    int max = 0;
    for(int i=1; i<nbs.length; i++) {
    	if(nbs[i]>max) {
    		max = nbs[i];
    	}
    }
    return max;
  }
}
\end{Java}
\end{frame}

\begin{frame}[fragile]{JUnit - Exemple (suite)}
\begin{Java}
package be.he2b.esi.java1;

import org.junit.Test; 
import static org.junit.Assert.*; 

public class OutilTest {
	
    @Test
    public void maxCas1Général() {
    	int[] nbs = {1, 3, 0};
    	assertEquals(3, Outil.max(nbs));
    }
    
    @Test
    public void maxCas2MaxDébut() {
    	int[] nbs = {1, -3, -4};
    	assertEquals(1, Outil.max(nbs));
    }
\end{Java}
\end{frame}

\begin{frame}[fragile]{JUnit - Exemple (fin)}

    	int[] nbs = {-1, -3, -4};

\begin{frame}[fragile]{JUnit - Tester les exceptions}
	\begin{itemize}
		\item 
			On ne peut pas tester la valeur de retour\dots
			\\Il n'y en a pas !
		\item	
			Spécifié via un \emph{paramètre} de l'\textbf{annotation}
			\medskip
			\begin{Java}
    @Test(expected = IllegalArgumentException.class)
    public void maxCas5AucunNombre() {
    	int[] nbs = {};
    	Outil.max(nbs);
    }
			\end{Java}
		\item Le test est réussi si la méthode lance l'\emph{exception} indiquée
		\item Ce qui n'est pas le cas dans notre exemple. Pourquoi ?
	\end{itemize}

\subsection{Conclusion}

\full[bluepigment]{

		\og\,Faut-il \textit{tout} tester ?\,\fg
	\end{center}

\note{
	\emph{\og Il faut vraiment tout tester ? \fg}
	\begin{itemize}
		\item Compromis entre le temps que l'on consacre aux tests et 
			la probabilité de trouver une erreur
		\item Trop simple $\longrightarrow$ perte de temps
		\item Attention ! On commet vite une erreur même dans du code simple
		\item De plus, les tests unitaires aident à la <<refactorisation>> 
			(cf. la leçon sur la lisibilité)
	\end{itemize}
}
