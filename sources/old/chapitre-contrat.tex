\subsection{Objectifs}

\begin{frame}{Objectif du cours}
\emph{S'initier à la programmation} au travers de l'apprentissage
\begin{itemize}
\item d'un langage (Java)
\item des bons comportements
  \begin{itemize} 
  \item Programmes lisibles, robustes, documentés et testés
  \item Capacités de \emph{déverminage} (debugging)
  \item Autonomie dans le travail \\(recherche dans la documentation)
  \end{itemize} 
\end{itemize}
\end{frame}

\subsection{Évaluations}

\begin{frame}{L'évaluation}
Deux \emph{cotes différentes}
\begin{itemize}
\item Une pour la partie \emph{cours}
\item Une autre pour la partie \emph{laboratoire}
\end{itemize}
\bigskip
\warning{On peut donc réussir une partie et pas l'autre.}
\end{frame}

\begin{frame}{L'évaluation du cours}
Uniquement un \emph{oral} en fin d'année
\\\bigskip
Qu'est-ce qu'on attend de vous ?
\begin{itemize}
\item \emph{Comprendre} les concepts abordés au cours 
 \\ (\emph{pas} de \emph{<<par coeur>>})
\item Pouvoir les expliquer clairement
\item Pouvoir les illustrer au travers d'exemples courts
\item De la \emph{rigueur} : pas d'<<à peu près>>
\item Du \emph{détail} : ne pas s'arrêter à l'<<usage courant>>
\end{itemize}
\end{frame}

\begin{frame}{L'évaluation des laboratoires}
\emph{Évaluations continues} pendant les laboratoires
\begin{itemize}
\item Courtes interrogations en \emph{début de séance}
\item Interrogations de \emph{synthèse}
\item Un \emph{projet}
\end{itemize}
\bigskip
Un examen en fin d'année permet de \emph{remettre en jeu} la cote d'année.
\\\bigskip
Qu'est-ce qu'on attend de vous ?
\begin{itemize}
\item De la \emph{rigueur}
\item De l'\emph{autonomie}
\end{itemize}
\end{frame}
