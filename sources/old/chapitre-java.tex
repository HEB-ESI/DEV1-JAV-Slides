\subsection{Historique}

\begin{frame}[fragile]{Historique de \sigle{Java}}
\begin{itemize}
\item [\emph{92}] \sigle{SUN} crée \sigle{oak} (systèmes embarqués).
\\Auteur: James Gosling
\item [\emph{94}] Adapté à Internet grâce aux \emph{applets}. 
\\Devient \sigle{Java}
\item [\emph{96}] Première version stable et gratuite de \sigle{JDK} 
\item [\emph{98}] Sortie de \sigle{Java 2}
\item [\emph{05}] Version \verb|1.5| de \sigle{Java 2}
\item [\emph{09}] \sigle{Oracle} rachète \sigle{Sun} (et donc \sigle{Java})
\item [\emph{11}] Version \verb|1.7| (Java 7, en GPL) \includegraphics[scale=.5]{../img/java7.jpeg}
\end{itemize}
\end{frame}

\subsection{Les éditions de \sigle{Java}}

\begin{frame}{Les éditions de \sigle{Java}}
\emph{Java} est disponible en 3 \emph{éditions}
  \begin{itemize}
  \item Même langage
  \item Mais taille de la bibliothèque différente
  \item Adapté à des situations différentes
  \end{itemize}
\bigskip
\emph{Java SE} (édition standard)
  \begin{itemize}
  \item Applications monopostes classiques
  \item Les applications Android sont en Java
  \end{itemize}
\end{frame}

\begin{frame}{Les éditions de \sigle{Java}}
\emph{Java ME} (édition mobile - plus léger) 
  \begin{itemize}
  \item Applications \emph{embarquées} : téléphones, appareils électroniques, \dots
  \item Omniprésent
  \end{itemize}
\medskip
\emph{Java EE} (édition entreprise - plus complet)
  \begin{itemize}
  \item Applications distribuées : client-serveur, web
  \item Très présent : riche, robuste et portable
  \item Concurrents : \sigle{.NET} (\sigle{Microsoft}), \sigle{PHP}
  \end{itemize}
\end{frame}

\subsection{Pourquoi \sigle{Java} ?}
 
\begin{frame}{Pourquoi \sigle{Java} ?}
Réelles \emph{qualités pédagogiques}  
  \begin{itemize}
  \item Syntaxe claire et précise
  \item Typage fort
  \item Détection précoce des erreurs
  \item Concepts modernes de programmation 
  \item Économie d'échelle
  \end{itemize}
\bigskip
A trouvé sa place dans le milieu professionnel
\end{frame}

\begin{frame}{\sigle{Java} et les autres langages}
Vous verrez d'autres langages
  \begin{itemize}
  \item Assembleur en première 
  \item \sigle{C} et \sigle{C++} en deuxième 
  \item \sigle{Cobol} en 1ère, 2ème et 3ème (Gestion)
  \end{itemize}
\bigskip
Vous approfondirez Java
  \begin{itemize}
  \item Les ateliers logiciels, applications distribuées
  \end{itemize}
\end{frame}
