\subsection{La notion de grammaire}

\begin{frame}{Le besoin d'une référence}
	Un langage de programmation doit pouvoir
		\begin{itemize}
		\item \emph{être décrit} auprès des programmeurs
		\item faire l'objet d'une \emph{compilation rigoureuse}
		\end{itemize} 
	\medskip\pause
	Il faut un \emph{document de référence}
		\begin{itemize}
		\item Pour Java : \emph{The Java Language Specification}
			\begin{itemize}
			\item Contient beaucoup de texte (en \emph{Anglais})
				\\ parfois \emph{incomplet} / \emph{ambigu}
			\item Nécessité d'utiliser un \emph{formalisme} précis : 
				\\ une \emph{grammaire} 
			\end{itemize} 
		\end{itemize} 
\end{frame}

\begin{frame}{Analogie avec le français}
	Le français aussi doit pouvoir être décrit
	\begin{center}
		phrase $\supset$ mots $\supset$ caractères
	\end{center} 
	\begin{itemize}
		\item
			L'\emph{alphabet} donne les caractères reconnus
		\item 
			Un \emph{dictionnaire/lexique} donne les mots corrects
		\item 
			La \emph{grammaire} renseigne sur l'ordre des mots
		\item 
			La \emph{sémantique} traite du sens de la phrase
		\item Exemples :
			\\\textit{<<L$\epsilon$ ch$\wedge\uparrow$ $\bowtie$>>}
			\hfill\textit{<<El tahc tse rion>>} 
			\\\textit{<<Le est chat noir>>} 
			\hfill\textit{<<La table boit la télé>>}
	\end{itemize}
\end{frame}

\begin{frame}{Un langage de programmation}
	Qu'en est-il pour un langage de programmation ?
	\begin{itemize}
		\item
			L'alphabet : liste donnée dans la référence
		\item 
			Les mots (\emph{token} en informatique)
			\begin{itemize}
				\item Impossible via un dictionnaire. Pourquoi ?
				\item Donnés par une \emph{grammaire lexicale}
			\end{itemize}
		\item 
			Les séquences de mots : données par une \emph{grammaire syntaxique}
		\item 
			La \emph{sémantique} : ne peut être décrite formellement
	\end{itemize}
\end{frame}

\begin{frame}{La notion de grammaire}
	Grammaire
		\begin{itemize}
		\item Description \emph{finie} de l'\emph{infinité} des programmes corrects
		\item Composée de
			\begin{itemize}
			\item l'ensemble des tokens pouvant apparaitre tels quels (\textit{symboles \emph{terminaux}})
			\item l'ensemble des \emph{règles de production}
				\begin{itemize}
				\item nom de la règle (\textit{symbole \emph{non terminal}})
				\item séquences de symboles produits par la règle 
				\end{itemize}
			\item une règle de production de \emph{départ} 
			\end{itemize}
		\end{itemize}
		\begin{center}
		Est \emph{valide} ce qui peut être \emph{produit} par la grammaire
		\end{center}
\end{frame}

\begin{frame}{La notion de grammaire}
	La grammaire \emph{lexicale} valide les tokens 
		\begin{itemize}
		\item les symboles terminaux sont les caractères
		\item les règles construisent le mot 
		\end{itemize}
	\bigskip
	La grammaire \emph{syntaxique} valide les séquences de tokens 
		\begin{itemize}
		\item les symboles terminaux sont les mots
		\item les règles construisent la phrase 
		\end{itemize}
\end{frame}

\begin{frame}[fragile]{Exemple : le langage MU}
Inventons un nouveau langage : le \emph{langage MU}
\begin{itemize}
\item Nous devons indiquer de manière précise quelle phrase est valide dans notre langage
\item Cela se fait en donnant sa grammaire
\pause
  \begin{itemize}
  \item Les 3 symboles terminaux: \texttt{M},\texttt{U},\texttt{I}
  \item Les 3 règles de production
  \begin{small}
\begin{verbatim}
  start:
       debut MU fin

  debut:                   fin:
       debut I                I fin
       I                      I 
\end{verbatim}
  \end{small}
  \item La règle de départ : \texttt{start}
  \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{Exemple : le langage MU}
Quelques \emph{phrases correctes} (qu'on peut \textit{produire}) 
  \begin{itemize}
  \item \texttt{IMUI}
  \item \texttt{IIMUI}
  \item \texttt{IMUIIIIIIIIIIIIIIIIIIIIIIIII}
  \item \dots à l'infini
  \end{itemize} 
\medskip \pause
Quelques \emph{phrases incorrectes} (impossibles à \textit{produire}) 
  \begin{itemize}
  \item \texttt{MU}
  \item \texttt{MUI}
  \item \texttt{MUIMU}
  \item \dots à l'infini
  \end{itemize} \end{frame}

\begin{frame}[fragile]{Exemple}
Voici un autre exemple
\begin{itemize}
\item Le symbole terminal: \texttt{A}
\item Les 2 règles de production
\begin{verbatim}
  liste:
       element
       element liste
  element:
       A  
\end{verbatim}
\item La règle de départ : \texttt{liste}
\end{itemize}
\bigskip
Quel est le langage produit par cette grammaire ?
\end{frame}

\subsection{Notations de la grammaire \sigle{Java}}

\begin{frame}[fragile]{Notations de la grammaire \sigle{Java}}
Une règle de la grammaire syntaxique java:
\begin{grammaire}
\nterm{ReturnStatement} :
     \term{return} Expression\opt;
\end{grammaire}
\begin{itemize}
\item Imaginez des exemples d'instruction \og return\fg\ valides
\end{itemize}
\bigskip
La \emph{grammaire complète} de \sigle{Java} 
peut être consultée dans le livre
\textit{\og The Java Language Specification\fg}
\end{frame}
