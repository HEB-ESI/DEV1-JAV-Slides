\begin{frame}[fragile]{Historique de \sigle{Java}}
\begin{itemize}
\item [\emph{92}] \sigle{SUN} crée \sigle{oak} (systèmes embarqués).
\\Auteur: James Gosling
\item [\emph{94}] Adapté à Internet grâce aux \emph{applets}. 
\\Devient \sigle{Java}
\item [\emph{96}] Première version stable et gratuite de \sigle{JDK} 
\item [\emph{98}] Sortie de \sigle{Java 2}
\item [\emph{05}] Version \verb|1.5| de \sigle{Java 2}
\item [\emph{09}] \sigle{Oracle} rachète \sigle{Sun} (et donc \sigle{Java})
\item [\emph{11}] Version \verb|1.7| (Java 7, en GPL)
\item [\emph{14}] Version \verb|1.8| (Java 8)
\end{itemize}
\end{frame}
