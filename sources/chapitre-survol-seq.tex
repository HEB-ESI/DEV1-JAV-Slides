\begin{frame}[fragile]{Définition d'une méthode}
	\textbf{Syntaxe}
	\bigskip
\begin{Java}
public static <typeRetour> nomMéthode ([typeDonnée1 nomDonnée1, 
		typeDonnée2 nomDonnée2, ...]) { 
    <<typeRetour> résultat>; 
    // instructions
    <return résultat>;  
}
\end{Java}
\bigskip
Si la méthode ne retourne rien \texttt{void}
\end{frame}

\begin{frame}[fragile]{Définition d'une méthode}
\emph{Exemple} : la valeur absolue
\begin{Java}
public static int absolu ( int nb ) {  
    int abs;
    if (nb<0) {
        abs = -nb;
    } else {
        abs = nb;
    }
    return abs;
}
\end{Java}
\begin{itemize}
\item Exemples d'appels dans la même classe
\begin{Java}
   int résultat = absolu(4);
   int écart = -10;
   int écartAbsolu = absolu(écart);
\end{Java}
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Définition d'une méthode}
	\emph{Exemple} : la moyenne de 2 réels

\begin{Java}
// Méthode qui calcule la moyenne de 2 nombres
// Données : 
// 	nb1 : le premier nombre (un réel)
// 	nb2 : le second nombre (un réel)
// Résultat : un réel qui représente la moyenne des 2 nombres

public static double moyenne ( double nb1, double nb2 ) {  
    double moyenne = (nb1 + nb2) / 2.0;
    return moyenne;
}
\end{Java}

	\begin{itemize}
		\item Appel possible (si dans la même classe)
\begin{Java}
  double cote = moyenne(34345, -3213213);
\end{Java}
	\end{itemize}
	
\end{frame}

\begin{frame}[fragile]{Exemple}
	\texttt{\small\$cat Moyenne.java}
\begin{Java}
  public class Moyenne {

    public static double moyenne(double nombre1, double nombre2) {
        return (nombre1 + nombre2) / 2;
    }
    
    public static void main(String[] args) {
      double moyenne;

      moyenne = moyenne(34345, -3213213);    
      System.out.println(moyenne); 
    }
  }
\end{Java}
\end{frame}

\begin{frame}[fragile]{Lire au clavier}
	\par\textit{Les applications modernes préfèrent les lectures dans des champs de saisies.}
	\par Dans une console, voici une manière de faire
	\medskip
	\par\emph{Exemple}
  \begin{Java}
  import java.util.Scanner;
  // ...
  Scanner clavier = new Scanner(System.in);
  // ...
  nombre1 = clavier.nextDouble();
  \end{Java}
\end{frame}

\begin{frame}[fragile]{Lire au clavier - Exemple}
	\texttt{\small\$cat Test.java}
\begin{Java}
import java.util.Scanner;

public class Test {
    public static double moyenne(double nombre1, double nombre2) {
        return (nombre1 + nombre2) / 2;
    }
    
    public static void main(String[] args) {
      Scanner clavier = new Scanner(System.in);
      double nombre1, nombre2;
      double moyenne;
      nombre1 = clavier.nextDouble();
      nombre2 = clavier.nextDouble();
      moyenne = moyenne(nombre1, nombre2);    
      System.out.println(moyenne); 
    }
}
\end{Java}
\end{frame}

\begin{frame}{Lire au clavier}
\begin{center}
\begin{tabular}{r|l}
Pour lire\dots & on écrit\dots \\ \hline
un entier & \java|nextInt()| \\ 
un réel & \java|nextDouble()| \\ 
un booléen & \java|nextBoolean()| \\ 
un mot & \java|next()|\\  
une ligne & \java|nextLine()|\\  
un caractère & \java|next().charAt(0)| \\ 
\end{tabular} 
\end{center}
\end{frame}

\begin{frame}[fragile]{Définition d'une méthode de lecture}

public static int lireEntier () {

\end{Java}

\end{Java}
\end{itemize}
\end{frame}

\subsection{Constantes}

\begin{frame}[fragile]{Constante locale}
	Une \textbf{constante} s'écrit grâce à \java|final| 
\par\medskip
\par\emph{Exemple}
\begin{Java}
  final int X = 1;
  final int Y;
  Y = 2*X;
  X = 2; // Erreur : possède déjà une valeur
  Y = 3; // Idem
\end{Java}
\end{frame}

\subsection{Conventions de noms}

\begin{frame}[fragile]{Conventions de noms}
Pour une variable :
\begin{itemize}
  \item Tout mettre en minuscules
  \item Sauf les débuts de noms composés en majuscule
\end{itemize}

Pour une constante :
\begin{itemize}
  \item Tout mettre en majuscules
  \item Utiliser \_ pour séparer les mots
\end{itemize}

Dans tous les cas : être explicite
\end{frame}
