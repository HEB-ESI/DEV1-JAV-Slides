\begin{frame}[fragile]{Instructions de choix}
Le \emph{Si}
\begin{Java}
  if ( condition ) {
    instructions
  }
\end{Java}
\bigskip
Le \emph{Si-sinon}
\begin{Java}
  if ( condition ) {
    instructions
  } else {
    instructions
  }
\end{Java} 
\end{frame}

\begin{frame}[fragile]{Exemple} 
\begin{Java}
import java.util.Scanner;
public class Test {
  public static void main(String[] args) {
      Scanner clavier = new Scanner(System.in);
      int nombre1;

      nombre1 = clavier.nextInt();
      System.out.println(positifOuNégatif(nombre1));
  }

  public static String positifOuNégatif(int nombre){     
      String chaine = nombre + " est un nombre ";
      if (nombre < 0) {
         chaine = chaine + "négatif"; 
      } else {
         chaine = chaine + "positif";   
      }
      return chaine;
  }
}
\end{Java}
\end{frame}

\begin{frame}[fragile]{Exercice} 
Comment traduire cet algorithme ?
\begin{Code}
Algorithme test (nombre1 : entier) \longrightarrow chaine
	chaineAfficher : chaine
	chaineAfficher \longleftarrow nombre1, "est "    
    Si nombre1 > 0 Alors
        chaineAfficher \longleftarrow chaineAfficher + "positif"
    Sinon
        Si nombre1 = 0 Alors
            chaineAfficher \longleftarrow chaineAfficher + "nul"
        Sinon
            chaineAfficher \longleftarrow chaineAfficher + "négatif"
        Fin Si
    Fin Si
    Retourner chaineAfficher
Fin Algorithme
\end{Code}
\end{frame}

\begin{frame}[fragile]{Le \og si-sinon-si\fg} 
L'exemple précédant s'écrirait mieux (extrait) :
\begin{Code}
    Si nombre1 > 0 Alors
        chaineAfficher \longleftarrow chaineAfficher + "positif"
    Sinon Si nombre1 = 0 Alors
        chaineAfficher \longleftarrow chaineAfficher + "nul"
    Sinon
        chaineAfficher \longleftarrow chaineAfficher + "négatif"
    Fin Si
\end{Code}
ce qui donnerait en Java
\begin{Java}
  if (nb>0) {
      chaineAfficher = chaineAfficher + "positif";
  } else if (nb==0) {
      chaineAfficher = chaineAfficher + "nul";
  } else {
      chaineAfficher = chaineAfficher + "négatif";
  }  
\end{Java}
\end{frame}

\begin{frame}[fragile]{Expressions booléennes} 
	Les comparateurs
	\begin{center}
		\Large
		\java|<| \; \java|>| \; \java|<=| \; \java|>=| \; \java|==| \; \java|!=|
	\end{center}
	Les opérateurs booléens 
	\begin{center}

\end{frame}

\begin{frame}[fragile]{Exemple}

import java.util.Scanner;

\begin{frame}[fragile]{Le \og selon-que\fg} 
\begin{Java}
  switch(produit) {
    case "Coca" : 
    case "Sprite" : 
    case "Fanta" :  
        prixDistributeur=60; 
        break;
    case "IceTea" : 
        prixDistributeur=70; 
        break;
    default :       
        prixDistributeur=0; 
        break;
  }
\end{Java}
\begin{itemize}
\item Notez le \java{break}
\item Possible avec : entiers, caractères et chaines
\end{itemize}
